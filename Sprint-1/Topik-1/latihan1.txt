// Langkah - langkah mengoperasikan ms word

$komputer =1;

while'$komputer <= 7' 
    DISPLAY '
        if'$komputer==1' 
            DISPLAY 'Pertama,nyalakan komputer';
        else if'$komputer==2' {
            DISPLAY 'Kedua,pilih ikon start atau tekan tombol star di keyboard';    
        else if'$komputer==3'
            DISPLAY 'Ketiga,pilih all program';
        else if'$komputer==4' 
            DISPLAY 'Keempat,pilih microsoft office';
        else if'$komputer==5' 
            DISPLAY 'Kelima,pilih microsoft office word';
        else if'$komputer==6'
            DISPLAY 'Keenam,pilih file';
        else
            DISPLAY 'Ketujuh,pilih new file';
        ENDIF
    
    '

}
